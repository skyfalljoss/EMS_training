function hashColor(name: string): string {
  let hash = 0
  for (let i = 0; i < name.length; i++) hash = name.charCodeAt(i) + ((hash << 5) - hash)
  return COLORS[Math.abs(hash) % COLORS.length] ?? COLORS[0]!
}

function formatDate(d: string | null | undefined): string {
  if (!d) return '—'
